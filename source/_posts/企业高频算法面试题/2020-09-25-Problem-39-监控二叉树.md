### 题目

​	https://leetcode.com/problems/binary-tree-cameras/

### 题目解析

- 贪心的思路，因为目标是很明确的，就是照完整棵树，相机树最少，那就是能不放则不放，主要考察两个小变量，放了没有?照到了没有？
- 结合二叉树的递归套路可解。

### 代码

```java
 public static class TreeNode {
      int val;
      TreeNode left;
      TreeNode right;
      TreeNode() {}
      TreeNode(int val) { this.val = val;}
      TreeNode(int val, TreeNode left, TreeNode right) {
          this.val = val;
          this.left = left;
          this.right = right;
      }
  }
    public static class Info{
        public boolean isMonitor; //节点放了没
        public int num;
        public boolean isSet;

        public Info(boolean isMonitor, int num, boolean isSet) {
            this.isMonitor = isMonitor;
            this.num = num;
            this.isSet = isSet;
        }
    }
    public static int minCameraCover(TreeNode root) {
        if(root.left == null && root.right ==null) return 1;
        final Info info = process1(root);
        return info.isMonitor == false ?info.num+1:info.num ;
    }

    //贪心解法
    private static Info process1(TreeNode node){
        if(node == null){
            return null;
        }
        boolean isSet = false,isMonitor = false;
        int num = 0;
        final Info leftInfo = process1(node.left);
        final Info rightInfo = process1(node.right);
        if(leftInfo == null && rightInfo == null){
            return new Info(false,0,false);
        }
        if(leftInfo != null && rightInfo == null){
            num += leftInfo.num;
            isSet = leftInfo.isMonitor == false;
            if(isSet) {
                num++;
                isMonitor = true;
            }
            if(leftInfo.isSet){
                isMonitor = true;
            }
        }
        else  if( rightInfo != null && leftInfo == null){
            num += rightInfo.num;
            isSet = rightInfo.isMonitor == false;
            if(isSet) {
                num++;
                isMonitor = true;
            }
            if(rightInfo.isSet)
                isMonitor = true;

        }
        else {
            num += leftInfo.num + rightInfo.num;
            isSet = (!leftInfo.isMonitor || !rightInfo.isMonitor);
            if(isSet){
                num++;
                isMonitor = true;
            }
            if(leftInfo.isSet || rightInfo.isSet){
                isMonitor = true;
            }
        }
        return new Info(isMonitor,num,isSet);
    }
```
