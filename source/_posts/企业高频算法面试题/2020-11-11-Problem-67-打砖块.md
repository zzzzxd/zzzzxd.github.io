### 题目

Leetcode原题：

https://leetcode.com/problems/bricks-falling-when-hit/

### 题目解析

- 首先一定要熟悉并查集。此题是并查集的逆运用，思考并查集核心思想在于集合的合并。而击落砖块问的是掉落之后的个数，相当于初始情况是一个大集合，不断的击落砖块时，集合分散之后，那些无连接的集合大小。所以可以，倒过来想，从击落最后一块砖块开始，此时是分散集合，不断的把击落的往回放，则相当于集合合并了。
- 涉及到连接天花板，我们可以设计天花板集合、天花板元素 这种变量。通过此题我们可以学会灵活设计变量这技巧。
- 小技巧:`cellSet[parent1]^cellSet[parent2]`判断是否相同

### 代码

```java
private static class UnionFind{
        public int parent[];
        public int cellSize;
        public boolean cellSet[];
        public int size[];
        public int[][] grid;
        public int M,N;

        public UnionFind(int[][] grid) {

            M = grid.length;
            N = grid[0].length;
            this.grid = grid;
            size = new int[M*N];
            parent = new int[M*N];
            cellSet = new boolean[M*N];
            for(int i = 0;i < M;i++){
                for(int j = 0;j < N;j++){
                    int index = i*N+j;
                    if(grid[i][j] == 1){
                        parent[index] = index;
                        size[index] = 1;
                        if(i == 0){
                            cellSet[index] = true;
                            cellSize++;
                        }
                    }
                }
            }
            initConnnet();
        }
        public void initConnnet(){
            for(int i = 0;i < M;i++){
                for (int j = 0;j < N;j++){ //直接上下左右的去连，边界处理，交给下游
                    union(i,j,i-1,j);
                    union(i,j,i+1,j);
                    union(i,j,i,j-1);
                    union(i,j,i,j+1);
                }
            }
        }
        public int find(int index){
            int par = parent[index];
            while (par != parent[par]){
                par = parent[par];
            }
            parent[index] = par;
            return par;
        }
        public void union(int r1,int c1,int r2,int c2){
            if(validLoc(r1,c1) && validLoc(r2,c2)){
                int loc1 = r1*N+c1;
                int loc2 = r2*N+c2;
                final int parent1 = find(loc1);
                final int parent2 = find(loc2);
                if(parent1 != parent2){
                    parent[parent1] = parent2;
                    if(cellSet[parent1]^cellSet[parent2]){
                        cellSize += cellSet[parent1] ? size[parent2]:size[parent1];
                        cellSet[parent2] = true;
                    }
                    size[parent2] += size[parent1];
                }
            }

        }
        public boolean validLoc(int r,int c) {
            return r >= 0 && r < M && c >= 0 && c < N && grid[r][c] == 1;
        }
        public int connect(int r,int c){
            grid[r][c] = 1;
            int cur = r*N+c;
            parent[cur] = cur;
            size[cur] = 1;
            int pre = cellSize;
            if(r == 0){ // 该恢复恢复，0行的肯定是天花板的元素,所以要是打的这里，本身就要先加入到天花板集合里
                cellSize++;
                cellSet[cur] = true;
            }
            union(r,c,r-1,c);
            union(r,c,r+1,c);
            union(r,c,r,c-1);
            union(r,c,r,c+1);
            int now = cellSize;
            return now == pre ? 0 : now-pre-1;
        }
    }

    public static int[] hitBricks(int[][] grid, int[][] hits) {
        int[] ans = new int[hits.length];
        int n = hits.length;
        if(n == 0) return ans;
        for(int i = 0;i < hits.length;i++){
            if(grid[hits[i][0]][hits[i][1]] == 1){
                grid[hits[i][0]][hits[i][1]] = 2;
            }
        }
        final UnionFind uf = new UnionFind(grid);
        for(int i = hits.length-1;i >= 0;i--){
            if(grid[hits[i][0]][hits[i][1]] == 2){
                ans[i] =  uf.connect(hits[i][0],hits[i][1]);
            }
        }
        return ans;
    }
    public static void main(String[] args) {
        final int[] ints = hitBricks(new int[][]{{1},{1},{1},{1},{1}}, new int[][]{{3,0},{4,0},{1,0},{2,0},{0,0}});
        System.out.println(Arrays.toString(ints));
    }
```
