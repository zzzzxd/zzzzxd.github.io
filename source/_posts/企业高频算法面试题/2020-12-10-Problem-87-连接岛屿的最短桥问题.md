### 题目

Leetcode 原题：

https://leetcode.com/problems/shortest-bridge/

### 题目解析

- 题目含义比较明确，就是有两座孤立的岛屿，然后看看连接起来 需要翻转几个0，显然bfs会很简单。

### 代码

```java
class Solution {
     public static int[] dirction_x = {-1,0,1,0};
    public static int[] dirction_y = {0,1,0,-1};

    public static int shortestBridge(int[][] grid) {
        int N = grid.length;
        //从一个1开始，记录下第一个岛屿的所有点,然后从这个岛屿进行bfs遍历，第一个遇到非同组的1，表明遇到另一个岛屿了
        final int[] group = new int[N*N];
        int i = 0,j = 0;
        for(i=0;i < N;i++){
            for (j = 0;j < N;j++){
                if(grid[i][j] == 1)
                    break;
            }
            if(j != N){
                break;
            }
        }
        //寻找其中一个连续的岛屿
        dfs(N,i,j,group,grid);
        //对这个连续对岛屿进行bfs 广播，一直到另一个岛屿
        return bfs(N, group, grid);


    }
    public static void dfs(int N,int row,int col,int[] group,int[][] grid){
        int index = row * N + col;
        if(!peek(N,row,col)||group[index] == 1 || grid[row][col] == 0){//已经访问过了,或者越界
            return;
        }
        group[index] = 1;
        for(int i = 0;i < 4;i++){
            dfs(N,row+dirction_x[i],col+dirction_y[i],group,grid);
        }
    }
    public static int bfs(int N,int[] group,int[][] grid){

        int[] queue = new int[N*N+10];
        int map[] = new int[N*N]; //记录岛屿到达此点的距离
        int visited[] = new int[N*N]; //代表是否入过队了
        int front  = 0,rear = 0;//rear虚指，front实指
        //初始化，岛屿全部元素入队,并且初始化距离为0
        for (int i = 0; i < group.length; i++) {
            if(group[i] == 1) {
                queue[rear++] = i;
                map[i] = 0;
                visited[i] = 1;
            }
        }
        while (rear != front){ //队列不空
             int index = queue[front++];
             int col = index % N;
             int row = index / N;
             //探测
             for(int i = 0;i < 4;i++){
                 int row_t =  row + dirction_x[i];
                 int col_t =  col + dirction_y[i];
                 int index_t = row_t * N + col_t;
                 if (peek(N,row_t,col_t)) {
                     if(visited[index_t] == 0){
                         if (grid[row_t][col_t] == 1) {
                             return map[index];
                         } else {
                             map[index_t] = map[index] + 1;
                             queue[rear++] = index_t;
                             visited[index_t] = 1;
                         }
                     }
                 }
             }
        }
        return 0;
    }
    //true代表不越界
    public static boolean peek(int N,int row,int col){
        return row >= 0 && row < N && col >= 0 && col < N;
    }
}
```
