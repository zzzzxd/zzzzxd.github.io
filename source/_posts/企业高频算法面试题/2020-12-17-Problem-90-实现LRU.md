### 题目

leetcode原题：

https://leetcode.com/problems/lru-cache/

### 题目解析

- 利用链表增删改的操作是 O(1)，哈希表可用来进行判断是否存在也是O(1)，结合起来。
- 可以利用双向链表 实现起来比较简单。重要的是维护好head、last变量。

### 代码

```java
public class LRUCache {
        private class LinkedNode{ 
            public int key;
            public int value;
            public LinkedNode next;
            public LinkedNode pre;
            public LinkedNode(int key, int value) {
                this.key = key;
                this.value = value;
            }
        }
        HashMap<Integer,LinkedNode> data;
        LinkedNode head;
        LinkedNode last;
        int count = 0;
        int capacity = 0;
        public LRUCache(int capacity) {
            data = new HashMap<>();
            this.capacity = capacity;
        }

        public int get(int key) {
            int res = data.get(key) == null ? -1 : data.get(key).value;
            if(data.get(key) != null){
                final LinkedNode node = data.get(key);
                if(head != node){
                   evict(node);
                }
            }
            return res;
        }

        public void put(int key, int value) {
            //新增
            if(data.get(key) == null){
                if(count == capacity){
                    data.remove(last.key);
                    last = last.pre;
                    count--;
                }
                final LinkedNode node = new LinkedNode(key, value);
                node.key = key;
                node.value = value;
                if(count == 0){
                    head = node;
                    last = node;
                }else{
                    node.next = head;
                    head.pre = node;
                    head = node;
                }
                data.put(key,node);
                count++;
            }
            //更新
            else {
                final LinkedNode node = data.get(key);
                node.value = value;
                if(node == head){
                   return;
                }
                evict(node); //把原节点摘下，前置
            }
        }
        public void evict(LinkedNode node){
            if(last == node){ //最后一个节点
                last = last.pre;
                last.next = null;
            }
            else{//普通节点
                node.next.pre = node.pre;
                node.pre.next = node.next;
            }
            node.next = head;
            head.pre = node;
            head = node;
            head.pre = null;
        }

    public static void main(String[] args) {
        final LRUCache lruCache = new LRUCache(3);
        lruCache.put(1,1);
        lruCache.put(2,2);
        lruCache.put(3,3);
        lruCache.put(4,4);
        System.out.println(lruCache.get(4));
        System.out.println(lruCache.get(3));
        System.out.println(lruCache.get(2));
        System.out.println(lruCache.get(1));
        lruCache.put(5,5);
        System.out.println(lruCache.get(1));
        System.out.println(lruCache.get(2));
        System.out.println(lruCache.get(3)); 
        System.out.println(lruCache.get(4));
        System.out.println(lruCache.get(5));



    }
}
```
