### 题目

​	Leetcode原题：

https://leetcode.com/problems/scramble-string/

### 题目解析

- 思路比较直接，更加考察编程能力，包括这些位置的推断。以及递归的设计
- 普通递归，不去优化会卡在最后连个case上，加上缓存就可以顺利通过。主要是能快速写出来程序，所以没有进一步优化，也可以进一步写成数组严格递归的形式，甚至用斜率优化

### 代码

public static boolean isScramble(String s1, String s2) {
        final char[] chars1 = s1.toCharArray();
        final char[] chars2 = s2.toCharArray();
        if (s1.length() != s2.length()) return  false;
//        return process(chars1,0,chars2,0,s1.length());
        int[][][]dp = new int[s1.length()][s1.length()][s1.length()+1];
        return process1(chars1,0,chars2,0,s1.length(),dp);

    }
    static boolean process(char[] s1, int l1, char[] s2, int l2, int size){
        if(size == 1) return s1[l1] == s2[l2];
        for (int leftOffset = 0;leftOffset < size-1;leftOffset++){ //枚举交换点
            //交换
            if (isSameType(s1,l1,l1+leftOffset,s2,l2,l2+leftOffset) && isSameType(s1,l1+leftOffset+1,l1+size-1,s2,l2+leftOffset+1,l2+size-1)){
                if(process(s1,l1,s2,l2,leftOffset+1) && process(s1,l1+leftOffset+1,s2,l2+leftOffset+1,size-leftOffset-1)){
                    return true;
                }
            }
            if (isSameType(s1,l1,l1+leftOffset,s2,l2+size-1-leftOffset,l2+size-1) && isSameType(s1,l1+leftOffset+1,l1+size-1,s2,l2,l2+size-1-leftOffset-1)){
                if (process(s1,l1,s2,l2+size-1-leftOffset,leftOffset+1) && process(s1,l1+leftOffset+1,s2,l2,size-leftOffset-1)){
                    return true;
                }
            }
        }
        return false;
    }
    static boolean process1(char[] s1, int l1, char[] s2, int l2, int size,int[][][] dp){
        if(dp[l1][l2][size]!=0) return dp[l1][l2][size]==1;
        if(size == 1) return s1[l1] == s2[l2];
        for (int leftOffset = 0;leftOffset < size-1;leftOffset++){ //枚举交换点
            //交换
            if (isSameType(s1,l1,l1+leftOffset,s2,l2,l2+leftOffset) && isSameType(s1,l1+leftOffset+1,l1+size-1,s2,l2+leftOffset+1,l2+size-1)){
                if(process1(s1,l1,s2,l2,leftOffset+1,dp) && process1(s1,l1+leftOffset+1,s2,l2+leftOffset+1,size-leftOffset-1,dp)){
                    dp[l1][l2][size] = 1;
                    return true;
                }
            }
            if (isSameType(s1,l1,l1+leftOffset,s2,l2+size-1-leftOffset,l2+size-1) && isSameType(s1,l1+leftOffset+1,l1+size-1,s2,l2,l2+size-1-leftOffset-1)){
                if (process1(s1,l1,s2,l2+size-1-leftOffset,leftOffset+1,dp) && process1(s1,l1+leftOffset+1,s2,l2,size-leftOffset-1,dp)){
                    dp[l1][l2][size] = 1;
                    return true;
                }
            }
        }
        dp[l1][l2][size] = -1;
        return false;
    }
    static  boolean isSameType(char[] s1,int l1,int r1,char[] s2,int l2,int r2){
        int[]  map = new int[256];
        for (int i = l1; i <= r1 ; i++) {
            map[s1[i]]++;
        }
        for(int i = l2;i <= r2;i++){
            if(--map[s2[i]] < 0) return false;
        }
        return true;
    }
