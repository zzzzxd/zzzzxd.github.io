### 题目

Leetcode 题目 : 

https://leetcode.com/problems/word-ladder/

### 题目解析

- 只要能反应过来使用图的模型，就能做，只不过要麻烦些而已。

### 代码

```java
class Solution {
    public static String[] vToStr;
    public static HashMap<String,Integer> strToMap;
    //图论的好多东西还是的捡一捡的
    public static int ladderLength(String beginWord, String endWord, List<String> wordList) {
        int N = wordList.size()+1;
        vToStr = new String[N];
        strToMap =  new HashMap<>();
        wordList.add(beginWord);
        // 处理映射
        for (int i = 0; i < N; i++) {
            vToStr[i] = wordList.get(i);
            strToMap.put(vToStr[i],i);
        }
        final Integer end = strToMap.get(endWord);
        if(end == null) return 0;
        final List<List<Integer>> adjectTb = getAdjectTb();
        final int bfs = bfs(adjectTb,N,strToMap.get(beginWord),end);
        return bfs;
    }

    public static List<List<Integer>> getAdjectTb(){
        final List<List<Integer>> res = new ArrayList<>();
        for (int i = 0; i < vToStr.length; i++) {
            final String str = vToStr[i];
            final char[] chars = str.toCharArray();
            final ArrayList<Integer> list = new ArrayList<>();
            for (int i1 = 0; i1 < chars.length; i1++) {
                for (char k = 'a';k <= 'z';k++){
                    if(chars[i1] == k){
                        continue;
                    }
                    char tmp = chars[i1];
                    chars[i1] = k;
                    final String s = new String(chars);
                    final Integer con = strToMap.get(s);
                    if(con != null){
                        list.add(con);
                    }
                    chars[i1] = tmp;
                }
            }
            res.add(list);
        }
        return res;
    }
    public static int bfs(List<List<Integer>> adjectTb ,int N,int begin,int end){
        int[] queue = new int[N];
        int front = 0,rear = 0;
        queue[rear++] = begin;
        boolean[] visited = new boolean[N];
        int step = 1;
        visited[begin] = true;
        int batchSize = 1;
        while (front != rear){
            int next_batch = 0;
            for (int i = 0;i < batchSize;i++){
                int tmp = queue[front++];
                if(tmp == end)
                    return step;
                final List<Integer> adjents = adjectTb.get(tmp);
                for (int j = 0; j < adjents.size(); j++) {
                    final Integer t = adjents.get(j);
                    if(!visited[t]){
                        queue[rear++] = t;
                        visited[t] = true;
                        next_batch++;
                    }
                }
            }
            step++;
            batchSize = next_batch;
        }
        return 0;
    }
}
```
