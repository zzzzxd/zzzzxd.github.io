### 题目

https://leetcode.com/problems/maximum-sum-of-3-non-overlapping-subarrays/

### 题目解析

- 题目难在能不能想到构造辅助数组。
- 看看两个代码 。代码二要更简洁，思路更清晰。

### 代码

```java
public static int[] maxSumOfThreeSubarrays(int[] nums, int k) {
        int[] leftEnum = new int[nums.length]; //以i开头的K个和
        //left[i],从0~i上,长度为k的,最大子数组,right是从i~N-1上长度为k的最大子数组
        final int[] left = new int[nums.length];
        int[] lrange = new int[nums.length];
        final int[] right = new int[nums.length];
        int[] rrange = new int[nums.length];

        int sumK = 0;
        for (int i = 0; i < nums.length; i++) {
            if(i < k-1) sumK += nums[i]; //长度增长
            else{
                sumK += nums[i];
                if(i-1 < 0 || sumK > left[i-1]){
                    left[i] = sumK;
                    lrange[i] = i-k+1;
                }else{
                    left[i] = left[i-1];
                    lrange[i] = lrange[i-1];
                }
                sumK -= nums[i-k+1];
            }
        }
        sumK = 0;
        for (int i = nums.length-1;i >= 0;i--){
            if (i > nums.length-1 -k+1)
                sumK += nums[i];
            else{
                sumK +=nums[i];
                leftEnum[i] = sumK;
                if(i+1 > nums.length-1 || sumK >= right[i+1]){
                   right[i] = sumK;
                   rrange[i] = i;
                }else{
                    right[i] = right[i+1];
                    rrange[i] = rrange[i+1];
                }
                sumK -= nums[i+k-1];
            }
        }
        int ans = Integer.MIN_VALUE,a = 0,b = 0,c = 0;
        //枚举中间k个长度数组，两边查找最大的数
        for (int i = k;i < nums.length-2*k+1;i++){
            int tmpAns = leftEnum[i] + left[i-1] + right[i+k];
            if(tmpAns > ans){
                a = lrange[i-1];
                b = i;
                c = rrange[i+k];
                ans = tmpAns;
            }
        }
        return new int[]{a,b,c};
    }
```

==更优雅的代码==

```java
public static int[] maxSumOfThreeSubarrays(int[] nums, int k) {
		int N = nums.length;
		int[] range = new int[N]; // range[i]代表 以i开始的k长度子数组和
		int[] left = new int[N];  // left[i]代表 从0~i中最大k个长度子数组的开始位置
		int sum = 0;
		for (int i = 0; i < k; i++) {
			sum += nums[i];
		}
		range[0] = sum;
		left[k - 1] = 0;
		int max = sum;
		for (int i = k; i < N; i++) {
			sum = sum - nums[i - k] + nums[i];
			range[i - k + 1] = sum;
			left[i] = left[i - 1];
			if (sum > max) {
				max = sum;
				left[i] = i - k + 1;
			}
		}
		sum = 0;
		for (int i = N - 1; i >= N - k; i--) {
			sum += nums[i];
		}
		max = sum;
		int[] right = new int[N];   //right[i] 代表 i~length-1 中最大k个长度子数组开始位置
		right[N - k] = N - k;
		for (int i = N - k - 1; i >= 0; i--) {
			sum = sum - nums[i + k] + nums[i];
			right[i] = right[i + 1];
			if (sum >= max) {
				max = sum;
				right[i] = i;
			}
		}
		int a = 0;
		int b = 0;
		int c = 0;
		max = 0;
		for (int i = k; i < N - 2 * k + 1; i++) { // 中间一块的起始点 (0...k-1)选不了 i == N-1
			int part1 = range[left[i - 1]];
			int part2 = range[i];
			int part3 = range[right[i + k]];
			if (part1 + part2 + part3 > max) {
				max = part1 + part2 + part3;
				a = left[i - 1];
				b = i;
				c = right[i + k];
			}
		}
		return new int[] { a, b, c };
	}
```
