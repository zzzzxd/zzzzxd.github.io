### 题目

​	给定两个字符串s1和s2，问s2最少删除多少字符可以成为s1的子串？比如 s1 = "abcde"，s2 = "axbc"

### 题目解析

-  一种暴力思路，将s2进行挑选，挑选删除字符少的子序列 看是不是s1子串。
- 编辑距离思路，挑选出s1的所有子串，挨个进行编辑距离的dp

### 代码

```java
//解法一：暴力枚举，删除一个的时候，删除两个的时候，从小开始删，其实删完就是子序列，只要一个中，就中了
    public static int MyminCost1(String s1, String s2){
        final char[] chars = s2.toCharArray();
        for(int i = s2.length();i >= 1;i--){
            if(MyProcess1(i,s1,chars,0,"",new HashMap<String,Boolean>())) return s2.length()-i;
        }
        return s2.length();
    }
    public static boolean MyProcess1(int rest, String s1,char[] s2,int index, String subsquence, HashMap<String,Boolean> cache){

        if(rest > s2.length-index) return false; //必须挑完rest个，要固定从大到小枚举子序列，减枝
        if(cache.containsKey(subsquence) && !cache.get(subsquence)) return false;
        if(rest == 0) {
                if (cache.containsKey(subsquence)) return cache.get(subsquence);
                else {
                    final int res = s1.indexOf(subsquence);
                    cache.put(subsquence, res != -1);
                    return res != -1;
                }
            }
        boolean p1 = MyProcess1(rest-1,s1,s2,index+1,subsquence+s2[index],cache);
        boolean p2 = MyProcess1(rest,s1,s2,index+1,subsquence,cache);
        return p1 || p2;
    }

    //解法二：就是编辑问题的子问题，把s1的字串枚举出来，看看s2变成这个 花费多少删除代价
    public static int MyminCost2(String s1,String s2){

        int res = s2.length();
        for(int i = 0;i < s1.length();i++){
            for (int j = i;j < s1.length();j++){
                String subStr = s1.substring(i,j+1);
                int c = MyProcess2(subStr,s2);
                if(c == -1)
                    break;
                else
                    res = Math.min(res,c);
            }
        }
        return res;
    }

    private static int MyProcess2(String  s1, String s2) {
        final char[] arr1 = s1.toCharArray();
        final char[] arr2 = s2.toCharArray();
        int[][] dp = new int[s1.length()+1][s2.length()+1];
        dp[0][0] = 0;
        for(int i =1;i< s1.length()+1;i++)  dp[i][0] = -1;
        for(int j = 1;j<s2.length()+1;j++) dp[0][j] = j;
        for(int i = 1;i < s1.length()+1;i++){
            for (int j = 1;j < s2.length()+1;j++){
                int p1 = dp[i-1][j-1] !=-1 &&  arr1[i-1] == arr2[j-1]?dp[i-1][j-1]:Integer.MAX_VALUE;
                int p2 = dp[i][j-1]!=-1 ? dp[i][j-1] + 1 : Integer.MAX_VALUE;
                dp[i][j] = p1==Integer.MAX_VALUE&&p2==Integer.MAX_VALUE? -1:Math.min(p1,p2);
            }
        }
        return dp[s1.length()][s2.length()];
    }
```
